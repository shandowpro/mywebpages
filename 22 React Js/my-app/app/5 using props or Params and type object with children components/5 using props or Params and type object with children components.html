Practical script of [using props inside the child function with passed datatype from the defind type object] and passing props values from the main component function] & [ including children functions with the main function ] :

I] Define and include a single child using props passed through from the child component function into the main component function

  1. 1] Define dataytypes object [to use it's props with define the datatypes of component function's props | params ], as following syntax :
    type HeaderParams = {
    text : string
    }


  2. 2] Create Multiple children components function & with defining props with datatypes form the defind type object :
    // A] Define the first child function component {Header() component function } with define a prop with datatype of defined type object :
    function Header(props : HeaderParams ) {
    // using return to define [defined [prop] with one of the defined property in the [type object : HeaderParams ] :
    return (
    < div > {props.text} < /div >
    ) ;
    }


    // B] Define the main function component {Home() component function} with Including defined children components functions + passsing through value of defined prop in the child componet function :
    function Home() {
    // using return to include all page elements & defined children components functions :
    return (
    < main className = {style.main} >
    // Including normal Element within the main page :
    < p > Hello world < /p >

    // including defined child component function {Header()} into the main function , with passing through/assigning value of the defined [prop : text]
    < Header text='vlue of the prop with in the header' / >

    < /main >
    ) ;
    }

    // Exporting the main component function at the last of script [best practise] :
    export default Home ;





II] Define and include a multiple Nesting children components with passing [props , defined datatypes by using serveral type objects ] through from the child component function into another child component function ,and including all children components functions into the main component function

  1. 1] Define multiples dataytypes objects [to use it's in definging of props | params's datatypes of sevearal defined component function ], as following syntax :
    // Define First dataytype object (HeaderParams) [to use it's props with define the datatype of component function's props | params {Header()} ] , as following syntax :
    type HeaderParams = {
    text : string
    }


    // Define second datatype object (DescripParams) [to use it's props with define the datatypes of component function's props | params {Description} ], as following syntax :
    type DescripParams = {
    details : string
    }


    // Define third datatype object (ListParams) [to use it's props with define the datatypes of component function's props | params {List} ], as following syntax :
    type ListParams = {
    list : string
    }


  2. 2] Create Multiple Nested children components functions & with defining [props] with datatypes defined by type object :
    // A] Define the first child function component {Header() component function } with define a prop with datatype of defined type object [without using Nesting] & [with defining a props with datatype of defined type object (HeaderParams)] :
    function Header(props : HeaderParams ) {
    // using return to define [defined [prop] with with datatype of one of the defined property in the [type object : HeaderParams ] :
    return (
    < div > {props.text} < /div >
    ) ;
    }


    // B] Define the second child function component {List() component function } with define a prop with datatype of defined type object [without using Nesting] & [with defineing a props with datatype of defined type object (ListParams)] :
    function List(props : ListParams ) {
    // using return to define [defined [prop] with with datatype of one of the defined property in the [type object : ListParams ] :
    return (
    < div > {props.list} < /div >
    ) ;
    }


    // B] Define the second child function component {Description()} with define a prop with datatype defined by type object(DescripParams) [with using Nesting of including defined child component function {List} into this child component function {Description} ] :
    function Description (props : DescripParams ) {
    // using return to define [defined [prop] with one of the defined property in the [type object : DesripParams ] :
    return (
    // using {reactJs fragments} to include more than one child [single elment and another child fuction ] :
    < >

    // use the defined property [props] to define a property [with datatype by the one of type object [DescripParam : details] into this Child compononnet function (Description) :
    < div > {props.details} < /div >
    // Including another defined child function component [List] with assigning the value of passed prop of the included child :
    < List list = "This is the list property value " / >

    < >
    )
    }


    // C] Define the main function component {Home() component function} with Including all defined children components functions + passsing through value of defined props in the child component function :
    function Home() {
    // using return to include all page elements & defined children components functions + assigning the component props's values :
    return (
    < main className = {style.main} >
    // Including single Element :
    < p > Hello world < /p >

    // including {Header()} defined conponent function , with passing through the defined [prop : text]
    < Header text='vlue of the prop with in the header' / >

    // including {Description()} [which is a defined Nested Component function (Description ) ]
    < Description details = "This is the value of the passed prop (details) of the child function " / >

    < /main >
    ) ;
    }

    // Exporting the main component function at the last of script [best practise] :
    export default Home ;
